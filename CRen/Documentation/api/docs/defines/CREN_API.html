CREN_API

File: cren_defines.h

Type: C Macro

Description

Cross-platform library export/import macro for controlling symbol visibility in shared libraries (DLLs on Windows, SOs on Linux/macOS).
If CREN_SHARED_LIBRARY is defined, it builds as shared library, If not defined it builds as static library. 

#ifdef CREN_SHARED_LIBRARY
    #if defined(_MSC_VER)
        #if defined(CREN_BUILDING_DLL)
            #define CREN_API __declspec(dllexport)
        #else
            #define CREN_API __declspec(dllimport)
        #endif
    #else
        #define CREN_API __attribute__((visibility("default")))
    #endif
#else
    #define CREN_API
#endif
